     } else if (_questionType == 'sequence_builder') {
       // Sequence Builder format - validate sentences
       if (_sequenceSentences.length < 3) {
         ScaffoldMessenger.of(context).showSnackBar(
           const SnackBar(content: Text('Please add at least 3 sentences')),
         );
         setState(() => _isSaving = false);
         return;
       }
       
       // Check if all sentences are filled
       for (var sentence in _sequenceSentences) {
         if (sentence['controller']!.text.trim().isEmpty) {
           ScaffoldMessenger.of(context).showSnackBar(
             const SnackBar(content: Text('Please fill all sentences')),
           );
           setState(() => _isSaving = false);
           return;
         }
       }
       
       // Convert to sequence builder format
       final sentences = [];
       for (int i = 0; i < _sequenceSentences.length; i++) {
         sentences.add({
           'id': i + 1,
           'text': _sequenceSentences[i]['controller']!.text.trim(),
           'correct_position': i + 1,
         });
       }
       
       questionData['options'] = sentences;
